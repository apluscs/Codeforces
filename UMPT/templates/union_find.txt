// union find with rank


void set_union(int pa, int pb) {  // with rank
  if (pa == pb)
    return;
  if (rank[pa] == rank[pb])
    rank[pa]++; // tie-broke guaranteed
  if (rank[pa] > rank[pb])
    reps[pb] = pa;
  else
    reps[pa] = pb;
}

int find(int a) {
  if (reps[a] != -1){
    return reps[a] = find(reps[a]);  // with path compression;
  }
  return a;
}

// normal union find
void set_union(int a, int b) {
  int pa = find(a), pb = find(b);
  if (pa != pb) reps[pa] = pb;
}
